<script>

// -------------------------

// CONFIG (your sheet data)

// -------------------------


const rows = 10;

const cols = 5;


// Example dropdown values (customize freely)

const dropdownOptions = {

    "B2": ["Fish", "Coral", "Nudibranch"],

    "C2": ["Yes", "No"],

};


// Example read-only cells (like formulas)

const readOnlyCells = ["A1", "B1", "C1", "D1", "E1"];


// -------------------------

// BUILD GRID

// -------------------------


const sheet = document.getElementById("sheet");


function colLetter(i) {

    return String.fromCharCode(65 + i); // A-E

}


for (let r = 1; r <= rows; r++) {

    for (let c = 0; c < cols; c++) {


        let cellId = colLetter(c) + r;


        // HEADER ROW

        if (r === 1) {

            let div = document.createElement("div");

            div.className = "cell header";

            div.innerText = cellId;

            sheet.appendChild(div);

            continue;

        }


        // DROPDOWN CELL

        if (dropdownOptions[cellId]) {

            let select = document.createElement("select");

            select.className = "cell";


            dropdownOptions[cellId].forEach(opt => {

                let o = document.createElement("option");

                o.value = opt;

                o.innerText = opt;

                select.appendChild(o);

            });


            sheet.appendChild(select);

            continue;

        }


        // READ ONLY CELL

        if (readOnlyCells.includes(cellId)) {

            let div = document.createElement("div");

            div.className = "cell readonly";

            div.contentEditable = false;

            div.innerText = "";

            sheet.appendChild(div);

            continue;

        }


        // NORMAL EDITABLE CELL

        let input = document.createElement("input");

        input.className = "cell";

        input.setAttribute("data-cell", cellId);

        sheet.appendChild(input);

    }

}

</script>
